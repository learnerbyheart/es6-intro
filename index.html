Class


class Anwendungsmanager extends Mitarbeiter {
  constructor(name) { super(name); }

  static behebeIncident() { console.log("Löse HPSM Incident"); }

  get gruppe() { return this.gruppe; }

  set gruppe(gruppenName) { this.gruppe = gruppenName; }
}
const lucas = new Anwendungsmanager("Lucas");
lucas.gruppe = "DV";
assert.equals(lucas.gruppe, "DV");

Spread Operator


const personalien = {
  name: "Lucas",
  nachname: "Pradel"
};
const aenderung = {
  nachname: "Hirsch",
  alter: 30
};
const lucas = {
  ...personalien,
  aenderung
};
assert.equals(lucas, {name: "Lucas", nachname: "Hirsch", alter: 30});
